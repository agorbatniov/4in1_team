def calculate_average(grades):
    if not grades:
        return 0
    return sum(grades) / len(grades)

# –¢–µ—Å—Ç 1: –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ –æ—Ü–µ–Ω–∫–∞–º–∏

# –¢–µ—Å—Ç 2: –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å –æ–¥–Ω–æ–π –æ—Ü–µ–Ω–∫–æ–π

# –¢–µ—Å—Ç 3: –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å –ø—É—Å—Ç—ã–º —Å–ø–∏—Å–∫–æ–º –æ—Ü–µ–Ω–æ–∫ (–≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è 0)

# –¢–µ—Å—Ç 4: –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å —Ä–∞–∑–Ω—ã–º–∏ –æ—Ü–µ–Ω–∫–∞–º–∏
